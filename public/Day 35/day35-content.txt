How to handle SSL Certification in selenium?											VVVVVVVIIIIMMMPPPPPPPPPPPPPPPPPPPPPPP

		- SSL means Secure Socket Layers.
		- When we test application in QA Environment or UAT Environment, we use self signed SSL Certificate.
		- When we open application in browser then it shows """" Your Connection is not private OR this site is not secure""".
		- Selenium does not run script properly in this scenario we have explicit allow driver to accept the SSL Certificate
		- We add SSL certificate using setAcceptInsecureCerts(true) keyword.
		
Syntax:
		//create object of ChromeOptions class.
				ChromeOptions opt =new ChromeOptions();
		
		//use setAcceptInsecureCerts(true) method 
				opt.setAcceptInsecureCerts(true);

		//create object of ChromeDriver class by passing ChromeOptions instance name
				ChromeDriver driver =new ChromeDriver(opt);



step 1: create object of ChromeOptions class
			ChromeOptions opt =new ChromeOptions();

step 2: use setAcceptInsecureCerts() by passing true keyword

			opt.setAcceptInsecureCerts(true);

step 3: create object of ChromeDriver class by passing ChromeOptions instance name or object

			ChromeDriver driver =new ChromeDriver(opt);

step 4: use get() method to open a URL

		driver.get("https://self-signed.badssl.com/");
	


package Tutorial20;

import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;

public class Demo1 {

	public static void main(String[] args) {

		ChromeOptions opt = new ChromeOptions();

		opt.setAcceptInsecureCerts(true);

		ChromeDriver driver = new ChromeDriver(opt);

		driver.get("https://self-signed.badssl.com/");

	}

}

					ControlSetText("Title", "Text" , "ControlID", "path");

	- we click on object by passing Title of object , text of object, and control ID of object.
						ControlClick("Title ", "Text" , "ControlID");

package Tutorial20;

import java.io.IOException;

import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class Demo2 {

		ChromeDriver driver = new ChromeDriver();

		driver.get("https://demoqa.com/upload-download");

		WebElement wb = driver.findElement(By.xpath("//label[text()='Select a file']/following-sibling::input"));
		Actions act = new Actions(driver);
		act.click(wb).build().perform();

		Thread.sleep(3000);

		Runtime.getRuntime().exec("C:\\Users\\praf0\\OneDrive\\Desktop\\Upoad File folder\\UploadDocuments.exe");

	}

}
